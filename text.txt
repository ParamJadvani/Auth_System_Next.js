} from "@/components/ui/select";

                                        checked={form.watch(name) === 1}
                                        onCheckedChange={(checked) => form.setValue(name, checked ? 1 : 0)}

                                        <IconInput

                                        />

                                <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
